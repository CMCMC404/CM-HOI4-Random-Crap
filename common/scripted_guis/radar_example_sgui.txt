		context_type = player_context
        parent_window_token = politics_tab
		visible = {

		effects = {
            jojo_example_btn_click = {
                # toggle up visibility
                if = {
                    limit = {
                        has_country_flag = jojo_ui_open
                    }
                    clr_country_flag = jojo_ui_open
                }
                else = {
                    set_country_flag = jojo_ui_open
                }
                # *counter clockwise*, add your input to a input array
                add_to_array = { ROOT.jojo_stats_input = 67 } #B
                add_to_array = { ROOT.jojo_stats_input = 50 } #C
                add_to_array = { ROOT.jojo_stats_input = 16 } #E
                add_to_array = { ROOT.jojo_stats_input = 84 } #A
                add_to_array = { ROOT.jojo_stats_input = 67 } #B
                add_to_array = { ROOT.jojo_stats_input = 16 } #E
                # for loop from 0 to n where n equals the sides of you polygon
                for_loop_effect = {
                    start = 0 
                    end = 6
                    value = i
                    # j = i - 1 to locate the previous array item
                    # and when i = 0 you should set j to the last index of the input array instead
                    set_variable = { j = i }
					add_to_variable = { j = -1 }
                    if = {
						limit = { check_variable = { i = 0 } }
						set_variable = { left = ROOT.jojo_stats_input^5 }
					}
					else = {				
						set_variable = { left = ROOT.jojo_stats_input^j }				
					}
                    # get the current array item, times 100, then add it to the previous array item
                    set_variable = { right = ROOT.jojo_stats_input^i }
					multiply_variable = { right = 100 }
					set_variable = { ROOT.jojo_stats_output^i = left }
					add_to_variable = { ROOT.jojo_stats_output^i = right }
                }
            }
		}
		properties = { 
		}
